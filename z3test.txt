(declare-const t Real)
(declare-const h Real)
 (declare-const tp Real)
(declare-const hp Real)
(declare-const h1 Real)
(declare-const h2 Real)
(declare-const h3 Real)
(declare-const h4 Real)
(assert (= tp (+ t -10.0)))
(assert (= hp (+ h -6.0)))
(assert (> t 17.023387320299246))
(assert (< t 17.163387320299247))
(assert (> h 60.34700148820362))
(assert (< h 61.147001488203614))
(assert (= h1 (+ (* -55.1 tp) 1319.25)))
(assert (= h2 (+ (* -1.314 tp) 55.2857)))
(assert (= h3 (+ (* -6.3427 tp) 222.214)))
(assert (= h4 (+ (* -37.5 tp) 1032.3)))
(define-fun comfort () Bool
	(and (and (> hp h1) (> hp h2)) (and (< hp h3) (< hp h4)))
)
(assert (not comfort))
